重新排列数组

这题是easy，通过对i%2==0来判断是添加start或者mid

func shuffle(nums []int, n int) []int {
length := len(nums)
mid := length/2
start := 0
res := make([]int, 0)
for i := 0 ; i < length ; i++ {
if i % 2 == 0 i == 0 {
res = append(res, nums[start])
start++
}else {
res = append(res, nums[mid])
mid++
}
}
return res
}

数组中的 k 个最强值

首先题目给我们重新定义一个排序的方法，因此这道题实际上是考我们排序，所以这题可以用的排序方法有很多种，在一开始做这道题时，我使用了冒泡排序结果超时了，后来转为使用快慢指针的方法来进行排序。

func getStrongest(arr []int, k int) []int {
var ans = make([]int, k)
sort.Ints(arr)
n := len(arr)
mid := arr[(n-1)/2]
left, right := 0, n - 1
for i := 0; i < k; i++ {
hdiff, ldiff := arr[right]-mid, mid-arr[left]
if hdiff >= ldiff {
ans[i] = arr[right]
right–
} else {
ans[i] = arr[left]
left++
}
}
return ans
}
